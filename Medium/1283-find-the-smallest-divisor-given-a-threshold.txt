Time Complexity: O(n * log max(nums))
Space Complexity: O(1)

Code:
class Solution {
    public int smallestDivisor(int[] nums, int threshold) {
        int low=1;
        //find max(high)
        int high=nums[0];
        for(int i=0;i<nums.length;i++){
            if(nums[i]>high) high=nums[i];
        }
        //Binary search
        int smallestDivisor=1;
        while(low<=high){
            int mid=(low+high)/2;
            int sum=calculateSum(nums, mid);

            if(sum<=threshold) {
                smallestDivisor=mid;
                high=mid-1;
            }
            else {
                low=mid+1;
            }
        }
        return smallestDivisor;
    }

    //calculate sum with ceil value 
    public static int calculateSum(int[] nums, int mid){
        int sum=0;
        for(int i=0;i<nums.length;i++){
            sum+=Math.ceil( (double) nums[i]/mid);
        }
        return sum;
    }
}
